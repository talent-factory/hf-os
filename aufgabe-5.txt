6. Praktische Aufgabe: Dateisysteme (C)

Vervollständigter C-Code:
-------------------------

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <time.h>
#include <sys/stat.h>
#include <sys/types.h>
#include <fcntl.h>
#include <unistd.h>

void print_file_metadata(const char* filename) {
    struct stat file_info;

    // stat() aufrufen und auf Fehler prüfen
    if (stat(filename, &file_info) != 0) {
        perror("stat");
        return;
    }

    printf("Metadaten für Datei: %s\n", filename);
    printf("------------------------\n");

    printf("Dateigröße (Bytes):       %ld\n", file_info.st_size);
    printf("Inode-Nummer:             %ld\n", file_info.st_ino);
    printf("Anzahl der Hardlinks:     %ld\n", file_info.st_nlink);
    printf("Zugriffsrechte (oktal):   %o\n", file_info.st_mode & 0777);

    char access_time[30];
    strftime(access_time, sizeof(access_time), "%Y-%m-%d %H:%M:%S", localtime(&file_info.st_atime));
    printf("Letzter Zugriff:          %s\n", access_time);

    char mod_time[30];
    strftime(mod_time, sizeof(mod_time), "%Y-%m-%d %H:%M:%S", localtime(&file_info.st_mtime));
    printf("Letzte Änderung:          %s\n", mod_time);

    printf("\n");
}

int main() {
    const char* filename = "testdatei.txt";
    const char* content = "Dies ist ein Test für Dateioperationen.\n"
                          "Wir untersuchen, wie Dateisysteme Metadaten verwalten.\n";

    // Datei erstellen und schreiben
    FILE* file = fopen(filename, "w");
    if (file == NULL) {
        perror("Fehler beim Öffnen der Datei");
        return 1;
    }

    fprintf(file, "%s", content);
    fclose(file);

    // Metadaten vor der Änderung anzeigen
    printf("Metadaten vor der Änderung:\n");
    print_file_metadata(filename);

    // Datei öffnen und Text anhängen
    file = fopen(filename, "a");
    if (file == NULL) {
        perror("Fehler beim Öffnen der Datei zum Anhängen");
        return 1;
    }

    fprintf(file, "Zusätzlicher Text zur Demonstration der Metadatenänderung.\n");
    fclose(file);

    // Metadaten nach der Änderung anzeigen
    printf("Metadaten nach der Änderung:\n");
    print_file_metadata(filename);

    // Hardlink erstellen
    const char* hardlink_name = "testdatei_hardlink.txt";
    if (link(filename, hardlink_name) != 0) {
        perror("Fehler beim Erstellen des Hardlinks");
    } else {
        printf("Hardlink erstellt: %s\n", hardlink_name);
    }

    // Metadaten des Hardlinks anzeigen
    printf("Metadaten des Hardlinks:\n");
    print_file_metadata(hardlink_name);

    // Dateien löschen
    remove(filename);
    remove(hardlink_name);

    return 0;
}

-------------------------
Analyse & Interpretation:
-------------------------

1. **Änderung der Metadaten nach dem Anhängen von Text:**
   - Die Dateigröße ist von **96 Bytes auf 157 Bytes** gestiegen.
   - Die **Inode-Nummer blieb gleich**, da es sich um dieselbe Datei handelt.
   - Interessant: Die **Zeitstempel (Zugriff + Änderung)** haben sich *nicht* verändert.
     → Das könnte auf ein Dateisystem-/OS-spezifisches Verhalten hinweisen (Caching, delayed updates).

2. **Inode vs. Dateiname:**
   - Der **Dateiname** ist ein Verzeichniseintrag.
   - Die **Inode-Nummer** identifiziert die Datei eindeutig im Dateisystem.
   - Zwei Dateinamen mit derselben Inode (wie `testdatei.txt` und `testdatei_hardlink.txt`) sind **Hardlinks**:
     → beide verweisen direkt auf denselben Speicherblock.

3. **Was bewirkt der Hardlink?**
   - Beide Dateien teilen sich nun die gleiche Inode **546372**.
   - `st_nlink` (Anzahl der Links) ist auf **2** gestiegen.
   - Wird **eine Datei gelöscht**, bleibt die andere erhalten → der Inhalt bleibt verfügbar, solange mind. 1 Link existiert.

4. **Was wäre anders bei einem symbolischen Link (Symlink)?**
   - Ein Symlink verweist **nicht** auf dieselben Daten, sondern nur auf den Pfad `testdatei.txt`.
   - Metadaten des Symlinks beziehen sich auf **den Link selbst**, nicht auf die Zieldatei.
   - Wird `testdatei.txt` gelöscht, ist der Symlink **gebrochen** → „Datei nicht gefunden“.
   - Symlinks haben **eigene Inode** und `st_nlink` bleibt **1**.
