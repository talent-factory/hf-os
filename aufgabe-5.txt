#include <stdio.h>
#include <stdlib.h>
#include <string.h>

void analyze_memory_layout(int* stack_var, int* heap_var, int* static_var) {
    printf("Speicheradressanalyse:\n");
    printf("---------------------\n");
    printf("Stack-Variable: %p\n", (void*)stack_var);
    printf("Heap-Variable: %p\n", (void*)heap_var);
    printf("Statische Variable: %p\n", (void*)static_var);

    // Differenzen berechnen
    unsigned long diff_stack_heap = (unsigned long)stack_var - (unsigned long)heap_var;
    unsigned long diff_stack_static = (unsigned long)stack_var - (unsigned long)static_var;
    printf("\nAdressdifferenzen:\n");
    printf("Stack - Heap: %lu\n", diff_stack_heap);
    printf("Stack - Statisch: %lu\n", diff_stack_static);

    printf("\nWachstumsrichtungen:\n");

    int local1;
    int local2;
    if (&local2 > &local1) {
        printf("Stack wächst nach oben\n");
    } else {
        printf("Stack wächst nach unten\n");
    }

    printf("\n");
}

int main() {
    static int static_variable = 100;
    int stack_variable = 200;

    int* heap_variable = (int*)malloc(sizeof(int));
    if (heap_variable == NULL) {
        printf("Fehler bei der Speicherzuweisung\n");
        return 1;
    }
    *heap_variable = 300;

    analyze_memory_layout(&stack_variable, heap_variable, &static_variable);

    for (int i = 0; i < 5; i++) {
        malloc(100); // Speicherleck – Speicher wird zugewiesen aber nie freigegeben
    }

    free(heap_variable);
    return 0;
}




// Ausgabeanalyse aufgabe-5

// Speicheradressanalyse zeigt drei unterschiedliche Adressbereiche
// Stack liegt bei 0x7ffd...
// Heap liegt bei 0x6366...
// Statische Variable liegt auch bei 0x6366 aber an anderer Stelle

// Adressdifferenz zeigt grosse Abstände zwischen Stack und den anderen Bereichen
// Das liegt daran dass Stack ganz oben im Adressraum liegt
// Heap und statische Variablen liegen weiter unten

// Die Richtung des Stackwachstums wird im Code durch Vergleich zweier lokaler Variablen geprüft
// Die Ausgabe zeigt Stack wächst nach oben
// Das ist eher ungewöhnlich denn in vielen Systemen wächst der Stack nach unten
// Erklärung könnte sein dass Compiler oder Optimierung Einfluss genommen hat oder die Adressvergabe zufällig so war

// Fazit
// Das Betriebssystem trennt die Speicherbereiche sauber
// Jeder Bereich hat seinen festen Zweck und wächst in eigene Richtung
// Speicherlecks wie im Code bleiben vom OS oft unbemerkt aber können langfristig Probleme machen
